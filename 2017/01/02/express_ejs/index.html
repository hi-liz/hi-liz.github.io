var express =require("express");

var mysql =require("mysql");

var multer =require("multer");
var upload = multer();

//安装的第三方的块无比要加载进来的呀，休要忘记的
var ejs =require("ejs");

//创建一个对象app 

var app =express();

//设置模板的路径

app.set("view engine","ejs");

app.use(express.static("public"));

//引用数据库中的数据

var connection = mysql.createConnection({
  host     : '127.0.0.1',
  user     : 'root',
  password : '',
  database : 'itcast_eric'
});

connection.connect();


//console.log(results);//是一个数组的形式

//这个要写在这里面，为什么要在这里，是，因为异步的原因，这个query 是异步的

var arr ;
app.get('/',function(req,res){
// res.send("hello world");
    connection.query('select * from menu', function (error, results, fields) {
        if (error) throw error;
        arr =results;
        //渲染模板
        res.render("index",{list:results});
    });


})

//也要放在这里面的
app.get('/delete',function(req,res){
//验证一下的是否做正确
// res.send("delete");
//根据req.query 可以直接得到utl参数的对象
//因为是字符串，所以要转换成number 类型
    var id =parseInt(req.query.id);


    for(var i =0 ;i<arr.length ;i++){
        if(arr[i].id ==id){
            console.log(id);
            connection.query('delete from menu where id ='+id+';', function (error, results, fields) {
                if (error) throw error;
                console.log(arr);
                res.redirect("/");
            });
        //找到后就可以直接跳出循环，结束函数
        //  break;
            //return;
        }
    }
})


app.get('/add',function(req,res){

    //先渲染页面的模板的
    res.render("add",{});
})

app.post('/add',upload.array(),function(req,res,next){
    // console.log(req)
    console.log(req.body);
    //多坑，下面的这个函数是异步的，这个ES6的方法的使用一定是这个符号``````
    //石油这个id 数据库会自动的给你添加的
    connection.query(`insert into menu(name,price) values("${req.body.title}","${req.body.content}");`, function (error, results, fields) {
      if (error) throw error;
        // console.log(results);

        res.redirect('/');
        next();
    });

})
